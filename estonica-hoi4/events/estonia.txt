country_event = {
	id = est.22
	title = est.22.t
	desc = est.22.d
	picture = GFX_report_event_physics_lab_01

	is_triggered_only = yes
	
	option = {
		name = est.22.a
		trigger = {
			NOT = { has_dlc = "Together for Victory" }
		}
		ai_chance = { base = 100 }
		add_political_power = -50
		add_tech_bonus = {
			name = "ships"
			bonus = 0.5
			uses = 1
			category = naval_equipment
		}
		EST = {
			add_tech_bonus = {
				name = industrial_bonus
				bonus = 0.5
				uses = 1
				category = industry
			}
			add_tech_bonus = {
				name = "ships"
				bonus = 0.5
				uses = 1
				category = naval_equipment
			}
			add_tech_bonus = {
				name = "heavy fighters"
				bonus = 0.5
				uses = 1
				category = cat_heavy_fighter
			}
			country_event = { days = 1 id = est.23 }
		}
	}
	
	option = {
		name = est.22.b
		trigger = { has_dlc = "Together for Victory" }
		ai_chance = { base = 100 }
		add_political_power = -50
		if = {
			limit = {
				EST = {
					OR = {
						is_in_faction_with = FIN
					}
					num_tech_sharing_groups < 1
				}
			}
			EST = { add_to_tech_sharing_group = EST_nordic_research }
		}
		if = {
			limit = {
				tag = FIN
				is_in_faction_with = EST
			}
			FIN = { add_to_tech_sharing_group = EST_nordic_research }
		}
		if = {
			limit = {
				tag = SWE
				is_in_faction_with = EST
			}
			SWE = { add_to_tech_sharing_group = EST_nordic_research }
		}
		if = {
			limit = {
				tag = NOR
				is_in_faction_with = EST
			}
			NOR = { add_to_tech_sharing_group = EST_nordic_research }
		}
		if = {
			limit = {
				tag = DEN
				is_in_faction_with = EST
			}
			DEN = { add_to_tech_sharing_group = EST_nordic_research }
		}
		EST = {
			country_event = { days = 1 id = est.23 }
		}
	}
	
	option = {
		name = est.22.c
		ai_chance = {
			base = 0
			modifier = {
				factor = 0
				has_opinion = {
					target = EST
					value > 90
				}
			}
		}
		EST = {
			country_event = { days = 1 id = est.24 }
		}
	}
}
